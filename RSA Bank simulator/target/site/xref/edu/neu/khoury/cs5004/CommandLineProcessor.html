4   /**
5    * Class CommandLineProcessor that makes sure arguments are correct.
6    */
7   public class CommandLineProcessor {
8     private static final Integer REQUIRED_ARGS = 4;
9   
10    /**
11     * Checks commandline argument if there are too many or few arguments.
12     * @param args - Arguments from commandline,
13     * @return - A secure bank verification object with all args from commandline,
14     * @throws IllegalArgumentException - If two few or to many args.
15     */
16    public static SecureBankVerificationSimulator processArguments(String[] args)
17        throws IllegalArgumentException {
18      if (args.length != REQUIRED_ARGS) {
19        throw new IllegalArgumentException("To few args");
20      }
21      return new SecureBankVerificationSimulator(Integer.valueOf(args[0]),
22          Integer.valueOf(args[1]), Double.valueOf(args[2]), args[3]);
23    }
